ğŸš€ Getting Started

Installation

npm install @guinetik/network-js

Basic Usage

import { NetworkStats } from '@guinetik/network-js';

// Define your network
const network = [
  { source: 'Alice', target: 'Bob', weight: 1 },
  { source: 'Bob', target: 'Charlie', weight: 2 },
  { source: 'Charlie', target: 'Alice', weight: 1 }
];

// Analyze it
const analyzer = new NetworkStats({ verbose: true });
const results = analyzer.analyze(network, ['degree', 'eigenvector']);

console.log(results);
// [
//   { id: 'Alice', degree: 2, eigenvector: 0.577 },
//   { id: 'Bob', degree: 2, eigenvector: 0.577 },
//   { id: 'Charlie', degree: 2, eigenvector: 0.577 }
// ]
