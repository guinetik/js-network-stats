âš¡ Strategy Pattern

Pass algorithm instances instead of strings for maximum flexibility and type safety. 

import { CommunityDetection, LouvainAlgorithm } from '@guinetik/network-js';

// Strategy Pattern: Pass algorithm instance
const louvain = new LouvainAlgorithm({ resolution: 1.0 });
const detector = new CommunityDetection(graph);
const result = detector.detectCommunities(louvain);

console.log(result.communities);    // { 'A': 0, 'B': 0, 'C': 1 }
console.log(result.modularity);     // 0.42
console.log(result.numCommunities); // 2

// Backward compatible: Pass string name
const result2 = detector.detectCommunities('louvain');

import { CommunityAlgorithm } from '@guinetik/network-js';

class MyAlgorithm extends CommunityAlgorithm {
  constructor(options = {}) {
    super('my-algorithm', 'My Custom Algorithm');
    this.options = options;
  }

  detect(graph) {
    // Your implementation here
    const communities = this._runAlgorithm(graph);

    return {
      communities,
      modularity: this._calculateModularity(graph, communities),
      numCommunities: new Set(Object.values(communities)).size,
      algorithm: this.name
    };
  }
}

// Use your custom algorithm
const myAlgo = new MyAlgorithm({ threshold: 0.5 });
const result = detector.detectCommunities(myAlgo);
